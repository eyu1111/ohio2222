-- 定义白名单
local whitelist = {"yukjm2"} -- 你可以在这里添加更多白名单用户
local player = game.Players.LocalPlayer
local isWhitelisted = false

-- 打印调试信息，确保玩家信息正确
print("Player Name: " .. player.Name)
print("Player UserId: " .. player.UserId)

-- 检查玩家是否在白名单中
for _, whitelistedUser in ipairs(whitelist) do
    print("Checking whitelist entry: " .. whitelistedUser)
    if player.Name == whitelistedUser or player.UserId == tonumber(whitelistedUser) then
        isWhitelisted = true
        break
    end
end

-- 打印调试信息，验证是否通过
if isWhitelisted then
    print("Player is whitelisted, loading script...")
    -- 白名单验证通过，加载你的脚本
    loadstring(game:HttpGet("https://raw.githubusercontent.com/eyu1111/ohio1111/main/Ohio111"))()
else
    print("Player is NOT whitelisted.")
    -- 白名单验证失败，弹出提示
    game.StarterGuiGui:SetCore("SendNotification", {
        Title = "验证失败",
        Text = "你不在白名单中，无法使用此脚本",
        Duration = 5
    })
end
